Generate raceline: ~/torcs_bin/bin/trackgen -c road -n g-track-2 -r
Generate shading: ~/torcs_bin/bin/accc +shad g-track-2.ac g-track-2-shade.ac
Generate acc: ~/torcs_bin/bin/accc -g g-track-2.acc -l0 g-track-2.ac -l1 g-track-2-shade.ac -l2 g-track-2-trk-raceline.ac -d3 1000 -d2 500 -d1 300 -S 300 -es
