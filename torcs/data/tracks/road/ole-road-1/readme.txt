Generate track: ~/torcs_bin/bin/trackgen -c road -n ole-road-1 -r
Generate raceline: ~/torcs_bin/bin/trackgen -c road -n ole-road-1 -r
Generate shading: ~/torcs_bin/bin/accc +shad ole-road-1.ac ole-road-1-shade.ac
Generate acc: ~/torcs_bin/bin/accc -g ole-road-1.acc -l0 ole-road-1.ac -l1 ole-road-1-shade.ac -l2 ole-road-1-trk-raceline.ac -d3 1000 -d2 500 -d1 300 -S 300 -es
